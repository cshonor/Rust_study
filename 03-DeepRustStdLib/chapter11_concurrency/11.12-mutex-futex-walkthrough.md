## A. 加锁快路径 + 慢路径

```rust
// 概念：std Mutex 内部 lock 逻辑
fn lock(&self) {
    loop {
        // 快路径：尝试 0 → 1
        match self.state.compare_exchange_weak(0, 1, AcqRel, Acquire) {
            Ok(_) => return, // 获锁，零 syscall
            Err(actual) => {
                // 慢路径：已锁或有竞争
                self.contended_lock(actual);
            }
        }
    }
}

fn contended_lock(&self, mut state: u32) {
    loop {
        // 标记存在等待者（示意：1 → 2 或类似编码）
        if state == 1 {
            match self.state.compare_exchange_weak(1, 2, Relaxed, Relaxed) {
                Ok(_) => {}
                Err(s) => { state = s; continue; }
            }
        }
        // 内核睡眠：仅当 state 仍为「已锁+等待」时才 block
        futex_wait(&self.state, 2); // SAFETY: 合法 futex 地址
        state = self.state.load(Acquire);
        // 被唤醒后再 CAS 抢锁
        match self.state.compare_exchange_weak(0, 1, AcqRel, Acquire) {
            Ok(_) => return,
            Err(s) => state = s,
        }
    }
}
```

## B. 解锁 + 唤醒

```rust
fn unlock(&self) {
    let prev = self.state.fetch_sub(1, Release); // 1→0 或 2→1
    if prev != 1 {
        // 有 waiter：重置并 wake
        self.state.store(0, Release);
        futex_wake(&self.state, 1); // 唤醒一个等待线程
    }
}
```

## C. 上层 poison + UnsafeCell（safe 边界）

```rust
// library/std/src/sync/mutex.rs — 结构示意
pub struct Mutex<T> {
    inner: sys::Mutex,
    poison: poison::Flag,
    data: UnsafeCell<T>,
}

pub fn lock(&self) -> LockResult<MutexGuard<'_, T>> {
    unsafe {
        self.inner.raw_lock(); // 内部 futex
    }
    // 检查 poison · 返回 MutexGuard
    // Guard Drop → raw_unlock()
}
```

| 层 | unsafe 在哪 |
